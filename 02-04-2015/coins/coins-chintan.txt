class Coins
	COINS = [10,5,2,1,0.50,0.25]
	def self.count(amount)
		coin_count = 0
		coin_count_recursive(amount,coin_count)		
	end	

	def self.coin_count_recursive(remaining_amount,coin_count)
		return coin_count if remaining_amount==0
		coin = COINS.select {|c| c <= remaining_amount}[0]
		coin_count += 1
		remaining_amount = remaining_amount - coin	
		coin_count_recursive(remaining_amount,coin_count)
	end

end
